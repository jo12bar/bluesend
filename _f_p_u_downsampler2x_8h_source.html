27 class Downsampler2xFPU

33  /*
34  Name: set_coefs
35  Description:
36  Sets filter coefficients. Generate them with the PolyphaseIir2Designer

38  Call this function before doing any processing.
39  Input parameters:
40  - coef_arr: Array of coefficients. There should be as many coefficients as

45  /*
46  Name: process_sample
47  Description:
48  Downsamples (x2) one pair of samples, to generate one output sample.
49  Input parameters:
50  - in_ptr: pointer on the two samples to decimate
51  Returns: Samplerate-reduced sample.
52  */
53  inline T process_sample(const T in_ptr [2]);
54 
55  /*
56  Name: process_block
57  Description:
58  Downsamples (x2) a block of samples.
59  Input and output blocks may overlap, see assert() for details.
60  Input parameters:
61  - in_ptr: Input array, containing nbr_spl * 2 samples.
62  - nbr_spl: Number of samples to output, > 0
63  Output parameters:
64  - out_ptr: Array for the output samples, capacity: nbr_spl samples.
65  */
66  void process_block(T out_ptr[], const T in_ptr[], long nbr_spl);
67 
68  /*
69  Name: process_sample_split
70  Description:
71  Split (spectrum-wise) in half a pair of samples. The lower part of the
72  spectrum is a classic downsampling, equivalent to the output of
73  process_sample().
74  The higher part is the complementary signal: original filter response
75  is flipped from left to right, becoming a high-pass filter with the same
76  cutoff frequency. This signal is then critically sampled (decimation by 2),
77  flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
78  Input parameters:
79  - in_ptr: pointer on the pair of input samples
80  Output parameters:
81  - low: output sample, lower part of the spectrum (downsampling)
82  - high: output sample, higher part of the spectrum.
83  */
84  inline void process_sample_split(T &low, T &high, const T in_ptr[2]);
85 
86  /*
87  Name: process_block_split
88  Description:
89  Split (spectrum-wise) in half a block of samples. The lower part of the
90  spectrum is a classic downsampling, equivalent to the output of
91  process_block().
92  The higher part is the complementary signal: original filter response
93  is flipped from left to right, becoming a high-pass filter with the same
94  cutoff frequency. This signal is then critically sampled (decimation by 2),
95  flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
96  Input and output blocks may overlap, see assert() for details.
97  Input parameters:
98  - in_ptr: Input array, containing nbr_spl * 2 samples.
99  - nbr_spl: Number of samples for each output, > 0
100  Output parameters:
101  - out_l_ptr: Array for the output samples, lower part of the spectrum
102  (downsampling). Capacity: nbr_spl samples.
103  - out_h_ptr: Array for the output samples, higher part of the spectrum.
104  Capacity: nbr_spl samples.
105  */
106  void process_block_split(T out_l_ptr[], T out_h_ptr[], const T in_ptr[], long nbr_spl);
107 
108  /*
109  Name: clear_buffers
110  Description:
111  Clears filter memory, as if it processed silence since an infinite amount

113  */
114  void clear_buffers();

177 template <int NC, typename T>
178 void Downsampler2xFPU <NC, T>::process_block (T out_ptr[], const T in_ptr[], long nbr_spl)
179 {
180  assert (in_ptr != 0);
181  assert (out_ptr != 0);
182  assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
183  assert (nbr_spl > 0);
184 
185  long pos = 0;
186  do
187  {
188  out_ptr [pos] = process_sample (&in_ptr [pos * 2]);
189  ++pos;
190  }
191  while (pos < nbr_spl);
192 }

194 template <int NC, typename T>
195 void Downsampler2xFPU <NC, T>::process_sample_split (T &low, T &high, const T in_ptr [2])
196 {
197  assert (&low != 0);
198  assert (&high != 0);
199  assert (in_ptr != 0);
200 
201  T spl_0 = in_ptr [1];
202  T spl_1 = in_ptr [0];
203 
204  #if defined (_MSC_VER)
205  #pragma inline_depth (255)
206  #endif // _MSC_VER
207 
208  StageProcFPU <NBR_COEFS, T>::process_sample_pos (
209  NBR_COEFS,
210  spl_0,
211  spl_1,
212  &_coef [0],
213  &_x [0],
214  &_y [0]
215  );
216 
217  low = (spl_0 + spl_1) * 0.5f;
218  high = spl_0 - low; // (spl_0 - spl_1) * 0.5f;
219 }

221 template <int NC, typename T>
222 void Downsampler2xFPU <NC, T>::process_block_split (T out_l_ptr[], T out_h_ptr[], const T in_ptr[], long nbr_spl)
223 {
224  assert (in_ptr != 0);
225  assert (out_l_ptr != 0);
226  assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
227  assert (out_h_ptr != 0);
228  assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
229  assert (out_h_ptr != out_l_ptr);
230  assert (nbr_spl > 0);
231 
232  long pos = 0;
233  do
234  {
235  process_sample_split
236  (
237  out_l_ptr [pos],
238  out_h_ptr [pos],
239  &in_ptr [pos * 2]
240  );
241  ++pos;
242  }
243  while (pos < nbr_spl);
244 }

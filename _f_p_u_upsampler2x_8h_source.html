137 template <int NC, typename T>
138 void Upsampler2xFPU <NC, T>::process_block (T out_ptr [], const T in_ptr [], long nbr_spl)
139 {
140  assert (out_ptr != 0);
141  assert (in_ptr != 0);
142  assert (out_ptr >= in_ptr + nbr_spl || in_ptr >= out_ptr + nbr_spl);
143  assert (nbr_spl > 0);
144 
145  long pos = 0;
146  do
147  {
148  process_sample (
149  out_ptr [pos * 2],
150  out_ptr [pos * 2 + 1],
151  in_ptr [pos]
152  );
153  ++ pos;
154  }
155  while (pos < nbr_spl);
156 }

55  FastSinOscillator - fast sinusoidal oscillator / table look up, based on an approach and code used by Miller Puckette in Pure Data, originally by Robert HÃ¶ldrich
56 
57  From some correspondence with M.S.P...
58 
59  The basic idea is this: if you have a double precision floating point number
60  between -2^19 (-524288) and +2^19 (+524288), and you want the fractional part as an
61  integer, proceed as follows:
62 
63  Add 3*2^19 (1572864) to it. The sum will be between 2^20 (1048576) and 2^21 (2097152) and so, in the usual
64  double precision format, the units bit will be the LSB of the higher-order
65  32 bits, and the fractional part will be neatly represented as an unsigned
66  fixed-point number between 0 and 2^32-1.
67 
68  You can then use a C union to read these lower 32 bits as an integer. Or, if
69  you want to access then as a floating-point number without having to use
70  fix-to-float conversion, just bash the upper 32 bits to their original
71  state for representing 3*2^19. So far this is what's going on in the wrap~
72  and phasor~ objects in Pd.
73 
74  The objects that do interpolating table lookup (e.g., cos~) take this one step
75  further. Here you end up with a number (now supposing it to be between
76  -2^10 and +2^10). Multiply by 2^9 (the size of the table) so that the
77  9-bit table address is in the 9 lowest bits of the upper 32 bit 'word', and
78  the fractional part is in the lower 32 bits as before. Grab the address as
79  an integer (masking to get just the 9 bits we want).
80 
81  The you need to re-convert the fractional address into floating-point format
82  so you can use it to interpolate. Just bash the upper 32 bits with the
83  original value for 3*2^19, and the result will be a 64-bit floating-point number
84  between 3*2^19 and 3*2^19+1. Subtracting 3*2^19 from this gives us a floating-
85  point representation of the fractional part.
86  */

126  static inline T Lookup(double phaseRadians)
127  {
128  double tPhase = phaseRadians / (PI * 2.) * tableSizeM1;
129 
130  tPhase += (double) UNITBIT32;
131 
132  union tabfudge tf;
133  tf.d = UNITBIT32;
134  const int normhipart = tf.i[HIOFFSET];
135 
136  tf.d = tPhase;
137  const T* addr = mLUT + (tf.i[HIOFFSET] & tableSizeM1);
138  tf.i[HIOFFSET] = normhipart;
139  const double frac = tf.d - UNITBIT32;
140  const T f1 = addr[0];
141  const T f2 = addr[1];
142  return f1 + frac * (f2 - f1);
143  }

145  void ProcessBlock(T* pOutput, int nFrames)
146  {
147  double phase = IOscillator<T>::mPhase + (double) UNITBIT32;
148  const double phaseIncr = IOscillator<T>::mPhaseIncr * tableSize;
149 
150  union tabfudge tf;
151  tf.d = UNITBIT32;
152  const int normhipart = tf.i[HIOFFSET];
153 
154  for (auto s = 0; s < nFrames; s++)
155  {
156  tf.d = phase;
157  phase += phaseIncr;
158  const T* addr = mLUT + (tf.i[HIOFFSET] & tableSizeM1); // Obtain the integer portion
159  tf.i[HIOFFSET] = normhipart; // Force the double to wrap.
160  const double frac = tf.d - UNITBIT32;
161  const T f1 = addr[0];
162  const T f2 = addr[1];
163  pOutput[s] = T(f1 + frac * (f2 - f1));
164  }
165 
166  // Restore mPhase
167  tf.d = UNITBIT32 * tableSize;
168  const int normhipart2 = tf.i[HIOFFSET];
169  tf.d = phase + (UNITBIT32 * tableSize - UNITBIT32); // Remove the offset we introduced at the start of UNITBIT32.
170  tf.i[HIOFFSET] = normhipart2;
171  IOscillator<T>::mPhase = tf.d - UNITBIT32 * tableSize;
172  }

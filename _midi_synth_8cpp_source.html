190 void MidiSynth::NoteOnOffPoly(const IMidiMsg& msg)
191 {
192  int status = msg.StatusMsg();
193  int velocity = msg.Velocity();
194  int note = msg.NoteNumber();
195 
196  if (status == IMidiMsg::kNoteOn && velocity)
197  {
198  velocity = Clip(mVelocityLUT[velocity], 1, 127);
199  double velNorm = (double) velocity / 127.;
200 
201  int v = FindFreeVoice(); // or first one triggered
202 
203  if (v == -1) // shouldn't happen
204  return;
205 
206  Voice* pVoice = GetVoice(v);
207 
208  if (!mHeldKeys.empty())
209  pVoice->mPrevKey = mHeldKeys.back().mKey;
210  else
211  pVoice->mPrevKey = note;
212 
213  KeyPressInfo theNote = KeyPressInfo(note, velNorm);
214 
215  if(std::find(mHeldKeys.begin(), mHeldKeys.end(), theNote) == mHeldKeys.end())
216  mHeldKeys.push_back(theNote);
217 
218  if(std::find(mSustainedNotes.begin(), mSustainedNotes.end(), theNote) == mSustainedNotes.end())
219  mSustainedNotes.push_back(theNote);
220 
221  pVoice->mStartTime = mSampleTime;
222  pVoice->mKey = note;
223  pVoice->mBasePitch = GetAdjustedPitch(note);
224  pVoice->mAftertouch = 0.;
225  pVoice->Trigger(theNote.mVelNorm, pVoice->GetBusy()); // if voice is busy it will retrigger
226 
227  mVoicesAreActive = true;
228  }
229  else // Note off
230  {
231  std::vector<KeyPressInfo>::iterator it;
232  bool erase = false;
233 
234  // REMOVE released key from held keys list. Do this even if the sustain pedal is down
235  for (it = mHeldKeys.begin(); it != mHeldKeys.end(); it++)
236  {
237  if (it->mKey == note)
238  {
239  erase = true;
240  break;
241  }
242  }
243

246 
247  if (!mSustainPedalDown)
248  {
249  // REMOVE released key from sustained keys list
250  erase = false;
251 
252  for (it = mSustainedNotes.begin(); it != mSustainedNotes.end(); it++)
253  {
